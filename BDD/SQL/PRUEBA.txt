CREATE TABLE CLIENTES (
	NOMBRE 					  VARCHAR2(15) NOT NULL,
	APELLIDOS 				VARCHAR2(30) NOT NULL,
	DNI						    CHAR(9) 	   NOT NULL,
	DIRECCION 				VARCHAR2(30),
	F_NACIMIENTO 			DATE,
	ESTADOCIVIL 			CHAR(1) 	   NOT NULL,
	ENTIDADBANCARIA 	CHAR(4),
	SUCURSAL 				  CHAR(4),
	DC 						    CHAR(2),
	CUENTA 					  CHAR(10),
	CONSTRAINT PK_CLIENTES PRIMARY KEY (DNI),
	CONSTRAINT CK_CLIENTES_ESTADOCIVIL CHECK (ESTADOCIVIL IN ('S','C','D','V'))
	);
CREATE TABLE SEGUROS (
	NUMEROPOLIZA 			CHAR(9) 	 NOT NULL,
	F_VENC					  DATE 		   NOT NULL,
	IMPORTE 				  NUMBER(6,2)NOT NULL,
	DURACION 				  CHAR(1) 	 NOT NULL,
	DNI			          CHAR(9)		NOT NULL,
	CONSTRAINT PK_SEGUROS PRIMARY KEY (NUMEROPOLIZA),
	CONSTRAINT FK_SEGUROS_CLIENTES FOREIGN KEY (DNI)
		REFERENCES CLIENTES ON DELETE CASCADE,
	CONSTRAINT CK_SEGUROS_DURACION CHECK (DURACION IN ('A','B','T','S','M'))
	);
