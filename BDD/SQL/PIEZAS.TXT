CREATE TABLE PROVEEDORES(
	NOMBRE			VARCHAR2(30),
	DIRECCION 		VARCHAR2(50)NOT NULL,
	POBLACION 		VARCHAR2(45)NOT NULL,
	PROVINCIA		VARCHAR2(22)NOT NULL,
	CODIGO_POSTAL 	CHAR(5)NOT NULL,
	CONSTRAINT PK_PROVEEDORES PRIMARY KEY (NOMBRE)
);
CREATE TABLE CATEGORIAS(
	NOMBRE	VARCHAR2(15),
	PRECIO	NUMBER(5,2) NOT NULL,
	CONSTRAINT PK_CATEGORIAS PRIMARY KEY (NOMBRE),
	CONSTRAINT CK_CATEGORIAS_PRECIO CHECK (PRECIO>0)
);
CREATE TABLE PIEZAS(
	COLOR 				VARCHAR2(10)NOT NULL,
	NOMBRE 				VARCHAR2(20),
	NOMBRE_CATEGORIA 	VARCHAR(15),
	CONSTRAINT PK_PIEZAS PRIMARY KEY (NOMBRE),
	CONSTRAINT FK_PIEZAS_CATEGORIAS FOREIGN KEY (NOMBRE_CATEGORIA)
		REFERENCES CATEGORIAS
);
CREATE TABLE SUMINISTRA(
	NOMBRE_PIEZAS 		VARCHAR2 (20),
	NOMBRE_PROVEEDORES  VARCHAR2 (30),
	CANTIDAD 			NUMBER(3)NOT NULL,
	CONSTRAINT FK_SUMINISTRA_PROVEEDORES FOREIGN  KEY (NOMBRE_PIEZAS)
		REFERENCES PROVEEDORES,
	CONSTRAINT FK_SUMINISTRA_PIEZAS FOREIGN KEY (NOMBRE_PIEZAS)
		REFERENCES PIEZAS,
	CONSTRAINT CK_SUMINISTRA_CANTIDAD CHECK (CANTIDAD>0)
);
