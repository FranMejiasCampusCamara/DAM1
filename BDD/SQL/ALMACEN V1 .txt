CREATE TABLE ESTANTERIAS (
	LETRAS 			VARCHAR2 (10) NOT NULL,
	NOMBRE			VARCHAR2 (20) NOT NULL,
	NUMERO_ALMACEN  NUMBER (1) NOT NULL,
	CONSTRAINT PK_ESTANTERIAS PRIMARY KEY (LETRAS),
	CONSTRAINT FK_ESTANTERIAS_ALMACENES FOREIGN KEY (NUMERO_ALMACEN)
	);

CREATE TABLE ALMACENES (
	NUMERO 			NUMBER (1) NOT NULL,
	DESCRIPCION		VARCHAR2 (30) NOT NULL,
	DIRECCION		VARCHAR2 (30) NOT NULL,
	POBLACION		VARCHAR2 (30) NOT NULL,
	PROVINCIA		VARCHAR2 (20) NOT NULL,
	CODIGO_POSTAL	VARCHAR2 (5) NOT NULL,
	CONSTRAINT PK_ALMACENES PRIMARY KEY (NUMERO)
	);

CREATE TABLE PIEZAS(
	TIPO_PIEZAS 		VARCHAR2 (10) NOT NULL,
	LETRAS_ESTANTERIA	VARCHAR2 (1) NOT NULL,
	NUMERO				NUMBER (1) NOT NULL,
	DESCRIPCION			VARCHAR2 (30) NOT NULL,
	PRECIO				NUMBER (4,2) NOT NULL,
	CONSTRAINT PK_PIEZAS PRIMARY KEY (NUMERO),
	CONSTRAINT FK_PIEZAS_TIPOPIEZA FOREIGN KEY (TIPO_PIEZAS),
	CONSTRAINT FK_PIEZAS_ESTANERIAS FOREIGN KEY (LETRAS_ESTANTERIA)
	);

CREATE TABLE ESTACOMPUESTA (
	CANTIDAD 		NUMBER (2) NOT NULL,
	NUMERO_PIEZAS 	NUMBER (1) NOT NULL,
	CONSTRAINT PK_ESTA_COMPUESTA PRIMARY KEY (NUMERO_PIEZAS),
	CONSTRAINT FK_ESTACOMPUESTA_PIEZAS FOREIGN KEY (NUMERO_PIEZAS)
	);

CREATE TABLE TIPOPIEZA (
	TIPO				VARCHAR2 (10) NOT NULL,
	DESCRIPCION 		VARCHAR2 (30) NOT NULL,
	CONSTRAINT PK_TIPOPIEZA PRIMARY KEY (TIPO)
	);
