CREATE TABLE TRENES (
	CODIGO	CHAR(3),
	MODELO	VARCHAR2(20),
	VELOCIDAD	CHAR (3),
	FECHA_ENTRADA	DATE,
	CONSTRAINT PK_TRENES PRIMARY KEY (CODIGO)
	);
CREATE TABLE LINEAS (
	CODIGO CHAR(2),
	NOMBRE VARCHAR2(30),
	COBERTURA VARCHAR2(20),
	CONSTRAINT PK_LINEAS PRIMARY KEY (CODIGO)
	);
CREATE TABLE ESTACIONES (
	CODIGO  CHAR(3),
	NOMBRE VARCHAR2(30),
	DISTRITO VARCHAR2(20)
	CONSTRAINT PK_ESTACIONES PRIMARY KEY (CODIGO)
	); 
CREATE TABLE TIENE_ASIGNADO (
	CODIGO_TREN CHAR(3),
	CODIGO_LINEA CHAR(2),
	CONSTRAINT PK_TIENE_ASIGNADO PRIMARY KEY (CODIGO_TREN)
